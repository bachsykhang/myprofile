Chuyển đổi danh sách cạnh sang ma trận kề

// chuyển đổi danh sách cạnh sang ma trận kề
package baitaptuan1;
import java.util.Scanner;
public class danhsachcanhsangmatranke {
public static void main(String[] args) {
Scanner sc = new Scanner(System.in);
int n, m;
// n là đỉnh, m là cạnh
int[][] a = new int[1001][1001];
System.out.print("Nhập số đỉnh: ");
n = sc.nextInt();
System.out.print("Nhập số cạnh: ");
m = sc.nextInt();
// duyệt cạnh
System.out.println("Danh sách cạnh");
for (int i = 0; i < m; i++) {
// nhập 2 đỉnh của cạnh đó
int x, y;
x = sc.nextInt();
y = sc.nextInt();
a[x][y] = a[y][x] = 1;
}
System.out.println("Ma trận kề");
for (int i = 1; i <= n; i++) {
for (int j = 1; j <= n; j++) {
System.out.print(a[i][j] + " ");
}
System.out.println();
}
}
}

Chuyển đổi danh sách cạnh sang danh sách kề

package baitaptuan1;
import java.util.*;
public class danhsachcanhsangdanhsachke {
public static void main(String[] args) {
Scanner sc = new Scanner(System.in);
int n, m;
ArrayList [] adj = new ArrayList[1001];
System.out.print("Nhập số đỉnh: ");
n = sc.nextInt();
System.out.print("Nhập số cạnh: ");
m = sc.nextInt();
// Khai báo bộ nhớ cho mảng
for (int i = 1; i <= 1000; i++) {
adj[i] = new ArrayList();
}
System.out.println("Danh sách cạnh");
// Nhập 2 đỉnh của cạnh đó
for (int i = 0; i < m; i++) {
int x, y;
x = sc.nextInt();
y = sc.nextInt();
adj[x].add(y);
adj[y].add(x);
}
// Đưa ra danh sách kề
System.out.println("Danh sách kề");
for (int i = 1; i <= n; i++) {
System.out.print(i + " :");
for (int x : adj[i]) {
System.out.print(x + " ");
}
System.out.println();
}
}
}

Lưu ý: Đối với ArrayList thêm integer sẽ có lỗi dấu"<" ">" nên các bạn coppy code nhớ check lại ở hình ảnh mà mình cung cấp.
